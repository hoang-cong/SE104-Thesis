\chapter{Giới thiệu đề tài}
    \section{Tổng quan đề tài}
    % Tìm hiểu hiện trạng kết hợp khảo sát.
    \paragraph{Trong thời đại quản lý số, việc kiểm soát hàng hóa thủ công gây ra nhiều sai sót và thất thoát. Đề tài "Quản lý nhà sách" được thực hiện nhằm xây dựng một công cụ hỗ trợ nhân viên và quản lý nhà sách thực hiện các nghiệp vụ hằng ngày một cách chính xác, nhanh chóng. Hệ thống tập trung vào việc quản lý dữ liệu tập trung, đảm bảo các quy định về tồn kho và khuyến mãi được thực thi tự động.}
    \paragraph{Các chức năng chính của phần mềm gồm có: Quản lý sách, quản lý kho, tra cứu sách, lập hóa đơn và lập báo cáo.}
    \section{Phạm vi đề tài}
        \subsection{Đối tượng nghiên cứu}
        \paragraph{Tập trung vào quy trình nghiệp vụ nội bộ của một nhà sách truyền thống, bao gồm các hoạt động: nhập kho, lưu trữ, bán hàng và báo cáo.}
        \subsection{Giới hạn của hệ thống}
        \begin{itemize}
            \item \textbf{Về chức năng:} Hệ thống chỉ tập trung vào quản lý bán hàng tại quầy, không bao gồm quản lý lương nhân viên hay tích hợp thanh toán qua ví điện tử.
            \item \textbf{Về công nghệ:} Ứng dụng được thiết kế chạy trên hệ điều hành Windows, cơ sở dữ liệu cài đặt tập trung tại máy chủ của cửa hàng.
            \item \textbf{Về người dùng:} Hệ thống phân quyền cho 4 nhóm đối tượng chính: Quản trị, Quản lý, Nhân viên và Thủ kho.
        \end{itemize}

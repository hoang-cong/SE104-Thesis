\chapter{Giới thiệu đề tài}
    \section{Tổng quan đề tài}
    % Tìm hiểu hiện trạng kết hợp khảo sát.
    \paragraph{Trong thời đại quản lý số, việc kiểm soát hàng hóa thủ công gây ra nhiều sai sót và thất thoát. Đề tài "Quản lý nhà sách" được thực hiện nhằm xây dựng một công cụ hỗ trợ nhân viên và quản lý nhà sách thực hiện các nghiệp vụ hằng ngày một cách chính xác, nhanh chóng. Hệ thống tập trung vào việc quản lý dữ liệu tập trung, đảm bảo các quy định về tồn kho và khuyến mãi được thực thi tự động.}
    \paragraph{Các chức năng chính của phần mềm gồm có: Quản lý sách, quản lý kho, tra cứu sách, lập hóa đơn và lập báo cáo.}
    \section{Phạm vi đề tài}
        \subsection{Đối tượng nghiên cứu}
        \paragraph{Tập trung vào quy trình nghiệp vụ nội bộ của một nhà sách truyền thống, bao gồm các hoạt động: nhập kho, lưu trữ, bán hàng và báo cáo.}
        \subsection{Giới hạn của hệ thống}
        \begin{itemize}
            \item \textbf{Về chức năng:} Hệ thống chỉ tập trung vào quản lý bán hàng tại quầy, không bao gồm quản lý lương nhân viên hay tích hợp thanh toán qua ví điện tử.
            \item \textbf{Về công nghệ:} Ứng dụng được thiết kế chạy trên hệ điều hành Windows, cơ sở dữ liệu cài đặt tập trung tại máy chủ của cửa hàng.
            \item \textbf{Về người dùng:} Hệ thống phân quyền cho 4 nhóm đối tượng chính: Quản trị, Quản lý, Nhân viên và Thủ kho.
        \end{itemize}

                \item Hóa đơn đi ra thiết bị xuất hoặc file gồm thành tiền, giảm giá và tổng tiền.
            \end{itemize}
            \paragraph{\textbf{Thuật toán:}}
            \begin{itemize}
                \item[] B1: Nhân viên bán hàng tạo hóa đơn mới và chọn các đầu sách khách mua.
                \item[] B2: Hệ thống kiểm tra QĐ6: hóa đơn phải có ít nhất một đầu sách.
                \item[] B3: Hệ thống đọc D1 Đầu sách để lấy đơn giá bán và đọc D2 Tồn kho để lấy số lượng tồn hiện tại.
                \item[] B4: Với từng dòng, hệ thống kiểm tra số lượng tồn có đủ để bán hay không.
                \item[] B5: Nếu tồn kho không đủ, hệ thống dừng xử lý dòng đó và hiển thị thông báo không cho phép bán.
                \item[] B6: Hệ thống đọc D3 Chương trình khuyến mại để tự động áp dụng giảm giá đang hiệu lực.
                \item[] B7: Hệ thống tính thành tiền từng dòng = số lượng * đơn giá * (1 - giảm giá).
                \item[] B8: Hệ thống tính tổng tiền hóa đơn và hiển thị cho nhân viên xác nhận.
                \item[] B9: Sau khi xác nhận, hệ thống lưu D4 Hóa đơn bán sách và chi tiết hóa đơn.
                \item[] B10: Hệ thống cập nhật D2 Tồn kho bằng cách trừ số lượng đã bán.
                \item[] B11: Hệ thống in hoặc xuất hóa đơn ra file.
            \end{itemize}

            \begin{figure}[H]

                \includegraphics[width=1\linewidth]{bm7___quan_ly_nguoi_dung.png}
            \end{figure}
            \paragraph{\textbf{Mô tả luồng dữ liệu:}}
            \begin{itemize}
                \item Quản lý gửi thông tin tài khoản hoặc yêu cầu thay đổi trạng thái người dùng.
                \item Tiến trình đọc vai trò hợp lệ, kiểm tra QĐ7 trên tên đăng nhập, mật khẩu và tài khoản quản lý duy nhất.
                \item Dữ liệu tài khoản hợp lệ được ghi vào kho người dùng, mật khẩu được lưu ở dạng băm.
                \item Kết quả trả về màn hình quản lý kèm trạng thái xử lý.
            \end{itemize}
            \paragraph{\textbf{Thuật toán:}}
            \begin{itemize}
                \item[] B1: Quản lý chọn thêm, sửa, khóa hoặc xóa tài khoản người dùng.
                \item[] B2: Hệ thống nhận họ tên, username, mật khẩu, vai trò và trạng thái tài khoản.
                \item[] B3: Hệ thống kiểm tra vai trò có thuộc các vai trò hợp lệ: Quản lý, Nhân viên bán hàng, Thủ kho.
                \item[] B4: Khi thêm hoặc sửa username, hệ thống kiểm tra QĐ7: username phải duy nhất trong D1 Người dùng.
                \item[] B5: Khi tạo hoặc đổi mật khẩu, hệ thống kiểm tra QĐ7: mật khẩu tối thiểu 6 ký tự, sau đó băm mật khẩu trước khi lưu.
                \item[] B6: Khi xóa tài khoản, hệ thống kiểm tra không được xóa tài khoản đang hoạt động hoặc tài khoản duy nhất có vai trò Quản lý.
                \item[] B7: Nếu vi phạm quy định, hệ thống trả lỗi và không thay đổi dữ liệu.
                \item[] B8: Nếu hợp lệ, hệ thống lưu thông tin tài khoản, vai trò, trạng thái vào D1 Người dùng.
                \item[] B9: Hệ thống ghi D3 Nhật ký thao tác và trả kết quả xử lý.
            \end{itemize}

                \item[] B2: Hệ thống kiểm tra tháng/năm hợp lệ và không vượt quá phạm vi dữ liệu.
                \item[] B3: Hệ thống truy vấn D1 Hóa đơn bán sách theo khoảng ngày đầu tháng đến cuối tháng.
                \item[] B4: Hệ thống đọc D2 Chi tiết hóa đơn để tính tổng số sách bán và doanh thu.
                \item[] B5: Hệ thống nhóm dữ liệu theo đầu sách để xác định top 10 sách bán chạy.
                \item[] B6: Hệ thống đọc D3 Tồn kho và D4 Đầu sách để lập tình trạng tồn kho cuối tháng.
                \item[] B7: Hệ thống tổng hợp các chỉ tiêu: tổng doanh thu, tổng số hóa đơn, tổng số sách đã bán, top sách và tồn kho.
                \item[] B8: Hệ thống hiển thị báo cáo và cho phép xuất PDF hoặc Excel.
            \end{itemize}

            \begin{figure}[H]
                \centering
                \includegraphics[width=1\linewidth]{qd9___thay_doi_quy_dinh.png}
            \end{figure}
            \paragraph{\textbf{Mô tả luồng dữ liệu:}}
            \begin{itemize}
                \item Quản lý nhập các thông số quy định mới như tồn tối thiểu, nhập tối thiểu và giảm giá tối đa.
                \item Tiến trình đọc giá trị cũ, kiểm tra miền giá trị hợp lệ rồi ghi giá trị mới vào kho quy định.
                \item Mọi thay đổi được ghi vào lịch sử với người thay đổi, thời điểm, giá trị cũ và giá trị mới.
                \item Sau khi lưu, các phân hệ nghiệp vụ đọc quy định mới và áp dụng ngay.
            \end{itemize}
            \paragraph{\textbf{Thuật toán:}}
            \begin{itemize}
                \item[] B1: Quản lý mở màn hình thay đổi quy định.
                \item[] B2: Hệ thống đọc D1 Quy định nghiệp vụ và hiển thị giá trị hiện tại.
                \item[] B3: Quản lý nhập giá trị mới cho số lượng tồn tối thiểu, số lượng nhập tối thiểu hoặc tỉ lệ giảm giá tối đa.
                \item[] B4: Hệ thống kiểm tra kiểu dữ liệu, miền giá trị và tính hợp lệ của từng thông số.
                \item[] B5: Hệ thống lưu lại giá trị cũ trước khi cập nhật.
                \item[] B6: Hệ thống ghi giá trị mới vào D1 Quy định nghiệp vụ và cho hiệu lực ngay sau khi lưu.
                \item[] B7: Hệ thống ghi D2 Lịch sử thay đổi quy định gồm người thay đổi, thời điểm, giá trị cũ và giá trị mới.
                \item[] B8: Hệ thống trả thông báo thành công; các nghiệp vụ BM1, BM2, BM3, BM5, BM6, BM7 sẽ đọc quy định mới khi xử lý.
            \end{itemize}
